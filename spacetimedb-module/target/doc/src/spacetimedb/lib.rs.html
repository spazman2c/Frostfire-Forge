50/// Generates code for registering a row-level security rule.
51///
52/// This attribute must be applied to a `const` binding of type [`Filter`].
53/// It will be interpreted as a filter on the table to which it applies, for all client queries.
54/// If a module contains multiple `client_visibility_filter`s for the same table,
55/// they will be unioned together as if by SQL `OR`,
56/// so that any row permitted by at least one filter is visible.
57///
58/// The `const` binding's identifier must be unique within the module.
59///
60/// The query follows the same syntax as a subscription query.
61///
62/// ## Example:
63///
64/// ```no_run
65/// # #[cfg(target_arch = "wasm32")] mod demo {
66/// use spacetimedb::{client_visibility_filter, Filter};
67///
68/// /// Players can only see what's in their chunk
69/// #[client_visibility_filter]
70/// const PLAYERS_SEE_ENTITIES_IN_SAME_CHUNK: Filter = Filter::Sql("
71///     SELECT * FROM LocationState WHERE chunk_index IN (
72///         SELECT chunk_index FROM LocationState WHERE entity_id IN (
73///             SELECT entity_id FROM UserState WHERE identity = :sender
74///         )
75///     )
76/// ");
77/// # }
78/// ```
79///
80/// Queries are not checked for syntactic or semantic validity
81/// until they are processed by the SpacetimeDB host.
82/// This means that errors in queries, such as syntax errors, type errors or unknown tables,
83/// will be reported during `spacetime publish`, not at compile time.
84#[cfg(feature = "unstable")]
85#[doc(inline, hidden)] // TODO: RLS filters are currently unimplemented, and are not enforced.
86pub use spacetimedb_bindings_macro::client_visibility_filter;

88/// Declares a table with a particular row type.
89///
90/// This attribute is applied to a struct type with named fields.
91/// This derives [`Serialize`], [`Deserialize`], [`SpacetimeType`], and [`Debug`] for the annotated struct.
92///
93/// Elements of the struct type are NOT automatically inserted into any global table.
94/// They are regular structs, with no special behavior.
95/// In particular, modifying them does not automatically modify the database!
96///
97/// Instead, a type implementing [`Table<Row = Self>`] is generated. This can be looked up in a [`ReducerContext`]
98/// using `ctx.db.{table_name}()`. This type represents a handle to a database table, and can be used to
99/// iterate and modify the table's elements. It is a view of the entire table -- the entire set of rows at the time of the reducer call.
100///
101/// # Example
102///
103/// ```ignore
104/// use spacetimedb::{table, ReducerContext};
105///
106/// #[table(name = user, public,
107///         index(name = popularity_and_username, btree(columns = [popularity, username])),
108/// )]
109/// pub struct User {
110///     #[auto_inc]
111///     #[primary_key]
112///     pub id: u32,
113///     #[unique]
114///     pub username: String,
115///     #[index(btree)]
116///     pub popularity: u32,
117/// }
118///
119/// fn demo(ctx: &ReducerContext) {
120///     // Use the name of the table to get a struct
121///     // implementing `spacetimedb::Table<Row = User>`.
122///     let user: user__TableHandle = ctx.db.user();
123///
124///     // You can use methods from `spacetimedb::Table`
125///     // on the table.
126///     log::debug!("User count: {}", user.count());
127///     for user in user.iter() {
128///         log::debug!("{:?}", user);
129///     }
130///
131///     // For every `#[index(btree)]`, the table has an extra method
132///     // for getting a corresponding `spacetimedb::BTreeIndex`.
133///     let by_popularity: RangedIndex<_, (u32,), _> =
134///         user.popularity();
135///     for popular_user in by_popularity.filter(95..) {
136///         log::debug!("Popular user: {:?}", popular_user);
137///     }
138///
139///     // There are similar methods for multi-column indexes.
140///     let by_popularity_and_username: RangedIndex<_, (u32, String), _> = user.popularity_and_username();
141///     for popular_user in by_popularity.filter((100, "a"..)) {
142///         log::debug!("Popular user whose name starts with 'a': {:?}", popular_user);
143///     }
144///
145///     // For every `#[unique]` or `#[primary_key]` field,
146///     // the table has an extra method that allows getting a
147///     // corresponding `spacetimedb::UniqueColumn`.
148///     let by_username: spacetimedb::UniqueColumn<_, String, _> = user.id();
149///     by_username.delete(&"test_user".to_string());
150/// }
151/// ```
152///
153/// See [`Table`], [`RangedIndex`], and [`UniqueColumn`] for more information on the methods available on these types.
154///
155/// # Browsing generated documentation
156///
157/// The `#[table]` macro generates different APIs depending on the contents of your table.
158///
159/// To browse the complete generated API for your tables, run `cargo doc` in your SpacetimeDB module project. Navigate to `[YOUR PROJECT/target/doc/spacetime_module/index.html` in your file explorer, and right click -> open it in a web browser.
160///
161/// For the example above, we would see three items:
162/// - A struct `User`. This is the struct you declared. It stores rows of the table `user`.
163/// - A struct `user__TableHandle`. This is an opaque handle that allows you to interact with the table `user`.
164/// - A trait `user` containing a single `fn user(&self) -> user__TableHandle`.
165///   This trait is implemented for the `db` field of a [`ReducerContext`], allowing you to get a
166///   `user__TableHandle` using `ctx.db.user()`.
167///
168/// # Macro arguments
169///
170/// The `#[table(...)]` attribute accepts any number of the following arguments, separated by commas.
171///
172/// Multiple `table` annotations can be present on the same type. This will generate
173/// multiple tables of the same row type, but with different names.
174///
175/// ### `name`
176///
177/// Specify the name of the table in the database. The name can be any valid Rust identifier.
178///
179/// The table name is used to get a handle to the table from a [`ReducerContext`].
180/// For a table *table*, use `ctx.db.{table}()` to do this.
181/// For example:
182/// ```ignore
183///  #[table(name = user)]
184///  pub struct User {
185///      #[auto_inc]
186///      #[primary_key]
187///      pub id: u32,
188///      #[unique]
189///      pub username: String,
190///      #[index(btree)]
191///      pub popularity: u32,
192///  }
193///  #[reducer]
194///  fn demo(ctx: &ReducerContext) {
195///      let user: user__TableHandle = ctx.db.user();
196///  }
197///  ```
198///
199/// ### `public` and `private`
200///
201/// Tables are private by default. This means that clients cannot read their contents
202/// or see that they exist.
203///
204/// If you'd like to make your table publicly accessible by clients,
205/// put `public` in the macro arguments (e.g.
206/// `#[spacetimedb::table(public)]`). You can also specify `private` if
207/// you'd like to be specific.
208///
209/// This is fully separate from Rust's module visibility
210/// system; `pub struct` or `pub(crate) struct` do not affect the table visibility, only
211/// the visibility of the items in your own source code.
212///
213/// ### `index(...)`
214///
215/// You can specify an index on one or more of the table's columns with the syntax:
216/// `index(name = my_index, btree(columns = [a, b, c]))`
217///
218/// You can also just put `#[index(btree)]` on the field itself if you only need
219/// a single-column index; see column attributes below.
220///
221/// A table may declare any number of indexes.
222///
223/// You can use indexes to efficiently [`filter`](crate::RangedIndex::filter) and
224/// [`delete`](crate::RangedIndex::delete) rows. This is encapsulated in the struct [`RangedIndex`].
225///
226/// For a table *table* and an index *index*, use:
227/// ```text
228/// ctx.db.{table}().{index}()
229/// ```
230/// to get a [`RangedIndex`] for a [`ReducerContext`].
231///
232/// For example:
233/// ```ignore
234/// let by_id_and_username: spacetimedb::RangedIndex<_, (u32, String), _> =
235///     ctx.db.user().by_id_and_username();
236/// ```
237///
238/// ### `scheduled(reducer_name)`
239///
240/// Used to declare a [scheduled reducer](macro@crate::reducer#scheduled-reducers).
241///
242/// The annotated struct type must have at least the following fields:
243/// - `scheduled_id: u64`
244/// - [`scheduled_at: ScheduleAt`](crate::ScheduleAt)
245///
246/// # Column (field) attributes
247///
248/// ### `#[auto_inc]`
249///
250/// Creates an auto-increment constraint.
251///
252/// When a row is inserted with the annotated field set to `0` (zero),
253/// the sequence is incremented, and this value is used instead.
254///
255/// Can only be used on numeric types.
256///
257/// May be combined with indexes or unique constraints.
258///
259/// Note that using `#[auto_inc]` on a field does not also imply `#[primary_key]` or `#[unique]`.
260/// If those semantics are desired, those attributes should also be used.
261///
262/// When `#[auto_inc]` is combined with a unique key,
263/// be wary not to manually insert values larger than the allocated sequence value.
264/// In this case, the sequence will eventually catch up, allocate a value that's already present,
265/// and cause a unique constraint violation.
266///
267/// ### `#[unique]`
268///
269/// Creates an unique constraint and index for the annotated field.
270///
271/// You can [`find`](crate::UniqueColumn::find), [`update`](crate::UniqueColumn::update),
272/// and [`delete`](crate::UniqueColumn::delete) rows by their unique columns.
273/// This is encapsulated in the struct [`UniqueColumn`].
274///
275/// For a table *table* and a column *column*, use:
276/// ```text
277/// ctx.db.{table}().{column}()`
278/// ```
279/// to get a [`UniqueColumn`] from a [`ReducerContext`].
280///
281/// For example:
282/// ```ignore
283/// let by_username: spacetimedb::UniqueColumn<_, String, _> = ctx.db.user().username();
284/// ```
285///
286/// When there is a unique column constraint on the table, insertion can fail if a uniqueness constraint is violated.
287/// If we insert two rows which have the same value of a unique column, the second will fail.
288/// This will be via a panic with [`Table::insert`] or via a `Result::Err` with [`Table::try_insert`].
289///
290/// For example:
291/// ```no_run
292/// # #[cfg(target_arch = "wasm32")] mod demo {
293/// use spacetimedb::{
294///     table,
295///     reducer,
296///     ReducerContext,
297///     // Make sure to import the `Table` trait to use `insert` or `try_insert`.
298///     Table
299/// };
300///
301/// type CountryCode = String;
302///
303/// #[table(name = country)]
304/// struct Country {
305///     #[unique]
306///     code: CountryCode,
307///     national_bird: String
308/// }
309///
310/// #[reducer]
311/// fn insert_unique_demo(ctx: &ReducerContext) {
312///     let result = ctx.db.country().try_insert(Country {
313///         code: "AU".into(), national_bird: "Emu".into()
314///     });
315///     assert!(result.is_ok());
316///
317///     let result = ctx.db.country().try_insert(Country {
318///         code: "AU".into(), national_bird: "Great Egret".into()
319///         // Whoops, this was Austria's national bird, not Australia's.
320///         // We should have used the country code "AT", not "AU".
321///     });
322///     // since there's already a country in the database with the code "AU",
323///     // SpacetimeDB gives us an error.
324///     assert!(result.is_err());
325///
326///     // The following line would panic, since we use `insert` rather than `try_insert`.
327///     // let result = ctx.db.country().insert(Country { code: "CN".into(), national_bird: "Blue Magpie".into() });
328///
329///     // If we wanted to *update* the row for Australia, we can use the `update` method of `UniqueIndex`.
330///     // The following line will succeed:
331///     ctx.db.country().code().update(Country {
332///         code: "AU".into(), national_bird: "Australian Emu".into()
333///     });
334/// }
335/// # }
336/// ```
337///
338/// ### `#[primary_key]`
339///
340/// Implies `#[unique]`. Also generates additional methods client-side for handling updates to the table.
341/// <!-- TODO: link to client-side documentation. -->
342///
343/// ### `#[index(btree)]`
344///
345/// Creates a single-column index with the specified algorithm.
346///
347/// It is an error to specify this attribute together with `#[unique]`.
348/// Unique constraints implicitly create a unique index, which is accessed using the [`UniqueColumn`] struct instead of the
349/// [`RangedIndex`] struct.
350///
351/// The created index has the same name as the column.
352///
353/// For a table *table* and an indexed *column*, use:
354/// ```text
355/// ctx.db.{table}().{column}()
356/// ```
357/// to get a [`RangedIndex`] from a [`ReducerContext`].
358///
359/// For example:
360///
361/// ```ignore
362/// ctx.db.cities().latitude()
363/// ```
364///
365/// # Generated code
366///
367/// For each `[table(name = {name})]` annotation on a type `{T}`, generates a struct
368/// `{name}__TableHandle` implementing [`Table<Row={T}>`](crate::Table), and a trait that allows looking up such a
369/// `{name}Handle` in a [`ReducerContext`].
370///
371/// The struct `{name}__TableHandle` is public and lives next to the row struct.
372/// Users are encouraged not to write the name of this table handle struct,
373/// or to store table handles in variables; operate through a `ReducerContext` instead.
374///
375/// For each named index declaration, add a method to `{name}__TableHandle` for getting a corresponding
376/// [`RangedIndex`].
377///
378/// For each field  with a `#[unique]` or `#[primary_key]` annotation,
379/// add a method to `{name}Handle` for getting a corresponding [`UniqueColumn`].
380///
381/// The following pseudocode illustrates the general idea. Curly braces are used to indicate templated
382/// names.
383///
384/// ```ignore
385/// use spacetimedb::{RangedIndex, UniqueColumn, Table, DbView};
386///
387/// // This generated struct is hidden and cannot be directly accessed.
388/// struct {name}__TableHandle { /* ... */ };
389///
390/// // It is a table handle.
391/// impl Table for {name}__TableHandle {
392///     type Row = {T};
393///     /* ... */
394/// }
395///
396/// // It can be looked up in a `ReducerContext`,
397/// // using `ctx.db().{name}()`.
398/// trait {name} {
399///     fn {name}(&self) -> Row = {T}>;
400/// }
401/// impl {name} for <ReducerContext as DbContext>::DbView { /* ... */ }
402///
403/// // Once looked up, it can be used to look up indexes.
404/// impl {name}Handle {
405///     // For each `#[unique]` or `#[primary_key]` field `{field}` of type `{F}`:
406///     fn {field}(&self) -> UniqueColumn<_, {F}, _> { /* ... */ };
407///
408///     // For each named index `{index}` on fields of type `{(F1, ..., FN)}`:
409///     fn {index}(&self) -> RangedIndex<_, {(F1, ..., FN)}, _>;
410/// }
411/// ```
412///
413/// [`Table<Row = Self>`]: `Table`

417/// Marks a function as a spacetimedb reducer.
418///
419/// A reducer is a function with read/write access to the database
420/// that can be invoked remotely by [clients].
421///
422/// Each reducer call runs in its own database transaction,
423/// and its updates to the database are only committed if the reducer returns successfully.
424///
425/// The first argument of a reducer is always a [`&ReducerContext`]. This context object
426/// allows accessing the database and viewing information about the caller, among other things.
427///
428/// After this, a reducer can take any number of arguments.
429/// These arguments must implement the [`SpacetimeType`], [`Serialize`], and [`Deserialize`] traits.
430/// All of these traits can be derived at once by marking a type with `#[derive(SpacetimeType)]`.
431///
432/// Reducers may return either `()` or `Result<(), E>` where [`E: std::fmt::Display`](std::fmt::Display).
433///
434/// ```no_run
435/// # #[cfg(target_arch = "wasm32")] mod demo {
436/// use spacetimedb::{reducer, SpacetimeType, ReducerContext};
437/// use log::info;
438/// use std::fmt;
439///
440/// #[reducer]
441/// pub fn hello_world(context: &ReducerContext) {
442///     info!("Hello, World!");
443/// }
444///
445/// #[reducer]
446/// pub fn add_person(context: &ReducerContext, name: String, age: u16) {
447///     // add a "person" to the database.
448/// }
449///
450/// #[derive(SpacetimeType, Debug)]
451/// struct Coordinates {
452///     x: f32,
453///     y: f32,
454/// }
455///
456/// enum AddPlaceError {
457///     InvalidCoordinates(Coordinates),
458///     InvalidName(String),
459/// }
460///
461/// impl fmt::Display for AddPlaceError {
462///     fn fmt(&self, f: &mut fmt::Formatter<'_>) -> Result<(), fmt::Error> {
463///         match self {
464///             AddPlaceError::InvalidCoordinates(coords) => {
465///                 write!(f, "invalid coordinates: {coords:?}")
466///             },
467///             AddPlaceError::InvalidName(name) => {
468///                 write!(f, "invalid name: {name:?}")
469///             },
470///         }
471///     }
472/// }
473///
474/// #[reducer]
475/// pub fn add_place(
476///     context: &ReducerContext,
477///     name: String,
478///     x: f32,
479///     y: f32,
480///     area: f32,
481/// ) -> Result<(), AddPlaceError> {
482///     // ... add a place to the database...
483///     # Ok(())
484/// }
485/// # }
486/// ```
487///
488/// Reducers may fail by returning a [`Result::Err`](std::result::Result) or by [panicking](std::panic!).
489/// Failures will abort the active database transaction.
490/// Any changes to the database made by the failed reducer call will be rolled back.
491///
492/// Reducers are limited in their ability to interact with the outside world.
493/// They do not directly return data aside from errors, and have no access to any
494/// network or filesystem interfaces.
495/// Calling methods from [`std::io`], [`std::net`], or [`std::fs`]
496/// inside a reducer will result in runtime errors.
497///
498/// Reducers can communicate information to the outside world in two ways:
499/// - They can modify tables in the database.
500///   See the `#[table]`(#table) macro documentation for information on how to declare and use tables.
501/// - They can call logging macros from the [`log`] crate.
502///   This writes to a private debug log attached to the database.
503///   Run `spacetime logs <DATABASE_IDENTITY>` to browse these.
504///
505/// Reducers are permitted to call other reducers, simply by passing their `ReducerContext` as the first argument.
506/// This is a regular function call, and does not involve any network communication. The callee will run within the
507/// caller's transaction, and any changes made by the callee will be committed or rolled back with the caller.
508///
509/// # Lifecycle Reducers
510///
511/// You can specify special lifecycle reducers that are run at set points in
512/// the module's lifecycle. You can have one of each per module.
513///
514/// These reducers cannot be called manually
515/// and may not have any parameters except for `ReducerContext`.
516///
517/// ### The `init` reducer
518///
519/// This reducer is marked with `#[spacetimedb::reducer(init)]`. It is run the first time a module is published
520/// and any time the database is cleared. (It does not have to be named `init`.)
521///
522/// If an error occurs when initializing, the module will not be published.
523///
524/// This reducer can be used to configure any static data tables used by your module. It can also be used to start running [scheduled reducers](#scheduled-reducers).
525///
526/// ### The `client_connected` reducer
527///
528/// This reducer is marked with `#[spacetimedb::reducer(client_connected)]`. It is run when a client connects to the SpacetimeDB module.
529/// Their identity can be found in the sender value of the `ReducerContext`.
530///
531/// If an error occurs in the reducer, the client will be disconnected.
532///
533/// ### The `client_disconnected` reducer
534///
535/// This reducer is marked with `#[spacetimedb::reducer(client_disconnected)]`. It is run when a client disconnects from the SpacetimeDB module.
536/// Their identity can be found in the sender value of the `ReducerContext`.
537///
538/// If an error occurs in the disconnect reducer,
539/// the client is still recorded as disconnected.
540///
541/// # Scheduled reducers
542///
543/// In addition to life cycle annotations, reducers can be made **scheduled**.
544/// This allows calling the reducers at a particular time, or in a loop.
545/// This can be used for game loops.
546///
547/// The scheduling information for a reducer is stored in a table.
548/// This table has two mandatory fields:
549/// - A primary key that identifies scheduled reducer calls.
550/// - A [`ScheduleAt`] field that says when to call the reducer.
551///
552/// Managing timers with a scheduled table is as simple as inserting or deleting rows from the table.
553/// This makes scheduling transactional in SpacetimeDB. If a reducer A first schedules B but then errors for some other reason, B will not be scheduled to run.
554///
555/// A [`ScheduleAt`] can be created from a [`spacetimedb::Timestamp`](crate::Timestamp), in which case the reducer will be scheduled once,
556/// or from a [`std::time::Duration`], in which case the reducer will be scheduled in a loop. In either case the conversion can be performed using [`Into::into`].
557///
558/// ```no_run
559/// # #[cfg(target_arch = "wasm32")] mod demo {
560/// use spacetimedb::{table, reducer, ReducerContext, Timestamp, TimeDuration, ScheduleAt, Table};
561/// use log::debug;
562///
563/// // First, we declare the table with scheduling information.
564///
565/// #[table(name = send_message_schedule, scheduled(send_message))]
566/// struct SendMessageSchedule {
567///     // Mandatory fields:
568///     // ============================
569///
570///     /// An identifier for the scheduled reducer call.
571///     #[primary_key]
572///     #[auto_inc]
573///     scheduled_id: u64,
574///
575///     /// Information about when the reducer should be called.
576///     scheduled_at: ScheduleAt,
577///
578///     // In addition to the mandatory fields, any number of fields can be added.
579///     // These can be used to provide extra information to the scheduled reducer.
580///
581///     // Custom fields:
582///     // ============================
583///
584///     /// The text of the scheduled message to send.
585///     text: String,
586/// }
587///
588/// // Then, we declare the scheduled reducer.
589/// // The first argument of the reducer should be, as always, a `&ReducerContext`.
590/// // The second argument should be a row of the scheduling information table.
591///
592/// #[reducer]
593/// fn send_message(ctx: &ReducerContext, arg: SendMessageSchedule) -> Result<(), String> {
594///     let message_to_send = arg.text;
595///
596///     // ... send the message ...
597///
598///     Ok(())
599/// }
600///
601/// // Now, we want to actually start scheduling reducers.
602/// // It's convenient to do this inside the `init` reducer.
603/// #[reducer(init)]
604/// fn init(ctx: &ReducerContext) {
605///
606///     let current_time = ctx.timestamp;
607///
608///     let ten_seconds = TimeDuration::from_micros(10_000_000);
609///
610///     let future_timestamp: Timestamp = ctx.timestamp + ten_seconds;
611///     ctx.db.send_message_schedule().insert(SendMessageSchedule {
612///         scheduled_id: 1,
613///         text:"I'm a bot sending a message one time".to_string(),
614///
615///         // Creating a `ScheduleAt` from a `Timestamp` results in the reducer
616///         // being called once, at exactly the time `future_timestamp`.
617///         scheduled_at: future_timestamp.into()
618///     });
619///
620///     let loop_duration: TimeDuration = ten_seconds;
621///     ctx.db.send_message_schedule().insert(SendMessageSchedule {
622///         scheduled_id: 0,
623///         text:"I'm a bot sending a message every 10 seconds".to_string(),
624///
625///         // Creating a `ScheduleAt` from a `Duration` results in the reducer
626///         // being called in a loop, once every `loop_duration`.
627///         scheduled_at: loop_duration.into()
628///     });
629/// }
630/// # }
631/// ```
632///
633/// Scheduled reducers are called on a best-effort basis and may be slightly delayed in their execution
634/// when a database is under heavy load.
635///
636/// ### Restricting scheduled reducers
637///
638/// Scheduled reducers are normal reducers, and may still be called by clients.
639/// If a scheduled reducer should only be called by the scheduler,
640/// consider beginning it with a check that the caller `Identity` is the module:
641///
642/// ```no_run
643/// # #[cfg(target_arch = "wasm32")] mod demo {
644/// use spacetimedb::{reducer, ReducerContext};
645///
646/// # #[derive(spacetimedb::SpacetimeType)] struct ScheduledArgs {}
647///
648/// #[reducer]
649/// fn scheduled(ctx: &ReducerContext, args: ScheduledArgs) -> Result<(), String> {
650///     if ctx.sender != ctx.identity() {
651///         return Err("Reducer `scheduled` may not be invoked by clients, only via scheduling.".into());
652///     }
653///     // Reducer body...
654///     # Ok(())
655/// }
656/// # }
657/// ```
658///
659/// <!-- TODO: SLAs? -->
660///
661/// [`&ReducerContext`]: `ReducerContext`
662/// [clients]: https://spacetimedb.com/docs/#client
663#[doc(inline)]
664pub use spacetimedb_bindings_macro::reducer;

666/// The context that any reducer is provided with.
667///
668/// This must be the first argument of the reducer. Clients of the module will
669/// only see arguments after the `ReducerContext`.
670///
671/// Includes information about the client calling the reducer and the time of invocation,
672/// as well as a view into the module's database.
673///
674/// If the crate was compiled with the `rand` feature, also includes faculties for random
675/// number generation.
676///
677/// Implements the `DbContext` trait for accessing views into a database.
678/// Currently, being this generic is only meaningful in clients,
679/// as `ReducerContext` is the only implementor of `DbContext` within modules.
680#[non_exhaustive]
681pub struct ReducerContext {
682    /// The `Identity` of the client that invoked the reducer.
683    pub sender: Identity,
684
685    /// The time at which the reducer was started.
686    pub timestamp: Timestamp,
687
688    /// The `ConnectionId` of the client that invoked the reducer.
689    ///
690    /// `None` if no `ConnectionId` was supplied to the `/database/call` HTTP endpoint,
691    /// or via the CLI's `spacetime call` subcommand.
692    ///
693    /// For automatic reducers, i.e. `init`, `client_connected`, `client_disconnected`, and scheduled reducers,
694    /// this will be the module's `ConnectionId`.
695    pub connection_id: Option<ConnectionId>,
696
697    /// Allows accessing the local database attached to a module.
698    ///
699    /// This slightly strange type appears to have no methods, but that is misleading.
700    /// The `#[table]` macro uses the trait system to add table accessors to this type.
701    /// These are generated methods that allow you to access specific tables.
702    ///
703    /// For a table named *table*, use `ctx.db.{table}()` to get a handle.
704    /// For example:
705    /// ```no_run
706    /// # mod demo { // work around doctest+index issue
707    /// # #![cfg(target_arch = "wasm32")]
708    /// use spacetimedb::{table, reducer, ReducerContext};
709    ///
710    /// #[table(name = book)]
711    /// #[derive(Debug)]
712    /// struct Book {
713    ///     #[primary_key]
714    ///     id: u64,
715    ///     isbn: String,
716    ///     name: String,
717    ///     #[index(btree)]
718    ///     author: String
719    /// }
720    ///
721    /// #[reducer]
722    /// fn find_books_by(ctx: &ReducerContext, author: String) {
723    ///     let book: &book__TableHandle = ctx.db.book();
724    ///
725    ///     log::debug!("looking up books by {author}...");
726    ///     for book in book.author().filter(&author) {
727    ///         log::debug!("- {book:?}");
728    ///     }
729    /// }
730    /// # }
731    /// ```
732    /// See the [`#[table]`](macro@crate::table) macro for more information.
733    pub db: Local,
734
735    #[cfg(feature = "rand08")]
736    rng: std::cell::OnceCell<StdbRng>,
737}
738
739impl ReducerContext {
740    #[doc(hidden)]
741    pub fn __dummy() -> Self {
742        Self {
743            db: Local {},
744            sender: Identity::__dummy(),
745            timestamp: Timestamp::UNIX_EPOCH,
746            connection_id: None,
747            rng: std::cell::OnceCell::new(),
748        }
749    }
750
751    /// Read the current module's [`Identity`].
752    pub fn identity(&self) -> Identity {
753        // Hypothetically, we *could* read the module identity out of the system tables.
754        // However, this would be:
755        // - Onerous, because we have no tooling to inspect the system tables from module code.
756        // - Slow (at least relatively),
757        //   because it would involve multiple host calls which hit the datastore,
758        //   as compared to a single host call which does not.
759        // As such, we've just defined a host call
760        // which reads the module identity out of the `InstanceEnv`.
761        Identity::from_byte_array(spacetimedb_bindings_sys::identity())
762    }
763}

774    /// Get a view into the tables.

799// #[cfg(target_arch = "wasm32")]
800// #[global_allocator]
801// static ALLOC: wee_alloc::WeeAlloc = wee_alloc::WeeAlloc::INIT;
802
803// This should guarantee in most cases that we don't have to reallocate an iterator
804// buffer, unless there's a single row that serializes to >1 MiB.
805const DEFAULT_BUFFER_CAPACITY: usize = spacetimedb_primitives::ROW_ITER_CHUNK_SIZE * 2;
806
807/// Queries and returns the `table_id` associated with the given (table) `name`.

809/// Panics if the table does not exist.

812    sys::table_id_from_name(table_name).unwrap_or_else(|_| {
813        panic!("Failed to get table with name: {table_name}");
814    })

817thread_local! {
818    /// A global pool of buffers used for iteration.
819    // This gets optimized away to a normal global since wasm32 doesn't have threads by default.
820    static ITER_BUFS: RefCell<Vec<Vec<u8>>> = const { RefCell::new(Vec::new()) };
821}
822
823struct IterBuf {
824    buf: Vec<u8>,
825}
826
827impl IterBuf {
828    /// Take a buffer from the pool of buffers for row iterators, if one exists. Otherwise, allocate a new one.
829    fn take() -> Self {
830        let buf = ITER_BUFS
831            .with_borrow_mut(|v| v.pop())
832            .unwrap_or_else(|| Vec::with_capacity(DEFAULT_BUFFER_CAPACITY));
833        Self { buf }
834    }
835
836    fn serialize<T: Serialize + ?Sized>(val: &T) -> Result<Self, bsatn::EncodeError> {
837        let mut buf = IterBuf::take();
838        buf.serialize_into(val)?;
839        Ok(buf)
840    }
841
842    #[inline]
843    fn serialize_into<T: Serialize + ?Sized>(&mut self, val: &T) -> Result<(), bsatn::EncodeError> {
844        bsatn::to_writer(&mut **self, val)
845    }
846}
847
848impl Drop for IterBuf {
849    fn drop(&mut self) {
850        self.buf.clear();
851        let buf = std::mem::take(&mut self.buf);
852        ITER_BUFS.with_borrow_mut(|v| v.push(buf));
853    }
854}
855
856impl AsRef<[u8]> for IterBuf {
857    fn as_ref(&self) -> &[u8] {
858        &self.buf
859    }
860}
861
862impl std::ops::Deref for IterBuf {
863    type Target = Vec<u8>;
864    fn deref(&self) -> &Self::Target {
865        &self.buf
866    }
867}
868impl std::ops::DerefMut for IterBuf {
869    fn deref_mut(&mut self) -> &mut Self::Target {
870        &mut self.buf
871    }
872}
