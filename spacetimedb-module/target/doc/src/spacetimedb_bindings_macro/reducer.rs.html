9#[derive(Default)]
10pub(crate) struct ReducerArgs {
11    name: Option<LitStr>,
12    lifecycle: Option<LifecycleReducer>,
13}
14
15enum LifecycleReducer {
16    Init(Span),
17    ClientConnected(Span),
18    ClientDisconnected(Span),
19    Update(Span),
20}
21impl LifecycleReducer {
22    fn to_lifecycle_value(&self) -> Option<TokenStream> {
23        let (Self::Init(span) | Self::ClientConnected(span) | Self::ClientDisconnected(span) | Self::Update(span)) =
24            *self;
25        let name = match self {
26            Self::Init(_) => "Init",
27            Self::ClientConnected(_) => "OnConnect",
28            Self::ClientDisconnected(_) => "OnDisconnect",
29            Self::Update(_) => return None,
30        };
31        let ident = Ident::new(name, span);
32        Some(quote_spanned!(span => spacetimedb::rt::LifecycleReducer::#ident))
33    }
34}
35
36impl ReducerArgs {
37    pub(crate) fn parse(input: TokenStream) -> syn::Result<Self> {
38        let mut args = Self::default();
39        syn::meta::parser(|meta| {
40            let mut set_lifecycle = |kind: fn(Span) -> _| -> syn::Result<()> {
41                check_duplicate_msg(&args.lifecycle, &meta, "already specified a lifecycle reducer kind")?;
42                args.lifecycle = Some(kind(meta.path.span()));
43                Ok(())
44            };
45            match_meta!(match meta {
46                sym::init => set_lifecycle(LifecycleReducer::Init)?,
47                sym::client_connected => set_lifecycle(LifecycleReducer::ClientConnected)?,
48                sym::client_disconnected => set_lifecycle(LifecycleReducer::ClientDisconnected)?,
49                sym::update => set_lifecycle(LifecycleReducer::Update)?,
50                sym::name => {
51                    check_duplicate(&args.name, &meta)?;
52                    args.name = Some(meta.value()?.parse()?);
53                }
54            });
55            Ok(())
56        })
57        .parse2(input)?;
58        Ok(args)
59    }
60}

62pub(crate) fn reducer_impl(args: ReducerArgs, original_function: &ItemFn) -> syn::Result<TokenStream> {
63    let func_name = &original_function.sig.ident;
64    let vis = &original_function.vis;
65
66    let reducer_name = args.name.unwrap_or_else(|| ident_to_litstr(func_name));
67
68    for param in &original_function.sig.generics.params {
69        let err = |msg| syn::Error::new_spanned(param, msg);
70        match param {
71            syn::GenericParam::Lifetime(_) => {}
72            syn::GenericParam::Type(_) => return Err(err("type parameters are not allowed on reducers")),
73            syn::GenericParam::Const(_) => return Err(err("const parameters are not allowed on reducers")),
74        }
75    }
76
77    let lifecycle = args.lifecycle.iter().filter_map(|lc| lc.to_lifecycle_value());
78
79    // Extract all function parameters, except for `self` ones that aren't allowed.
80    let typed_args = original_function
81        .sig
82        .inputs
83        .iter()
84        .map(|arg| match arg {
85            FnArg::Typed(arg) => Ok(arg),
86            _ => Err(syn::Error::new_spanned(arg, "expected typed argument")),
87        })
88        .collect::<syn::Result<Vec<_>>>()?;
89
90    // Extract all function parameter names.
91    let opt_arg_names = typed_args.iter().map(|arg| {
92        if let syn::Pat::Ident(i) = &*arg.pat {
93            let name = i.ident.to_string();
94            quote!(Some(#name))
95        } else {
96            quote!(None)
97        }
98    });
99
100    let arg_tys = typed_args.iter().map(|arg| arg.ty.as_ref()).collect::<Vec<_>>();
101    let first_arg_ty = arg_tys.first().into_iter();
102    let rest_arg_tys = arg_tys.iter().skip(1);
103
104    // Extract the return type.
105    let ret_ty = match &original_function.sig.output {
106        syn::ReturnType::Default => None,
107        syn::ReturnType::Type(_, t) => Some(&**t),
108    }
109    .into_iter();
110
111    let register_describer_symbol = format!("__preinit__20_register_describer_{}", reducer_name.value());
112
113    let lt_params = &original_function.sig.generics;
114    let lt_where_clause = &lt_params.where_clause;
115
116    let generated_describe_function = quote! {
117        #[export_name = #register_describer_symbol]
118        pub extern "C" fn __register_describer() {
119            spacetimedb::rt::register_reducer::<_, #func_name>(#func_name)
120        }
121    };
122
123    Ok(quote! {
124        const _: () = {
125            #generated_describe_function
126        };
127        #[allow(non_camel_case_types)]
128        #vis struct #func_name { _never: ::core::convert::Infallible }
129        const _: () = {
130            fn _assert_args #lt_params () #lt_where_clause {
131                #(let _ = <#first_arg_ty as spacetimedb::rt::ReducerContextArg>::_ITEM;)*
132                #(let _ = <#rest_arg_tys as spacetimedb::rt::ReducerArg>::_ITEM;)*
133                #(let _ = <#ret_ty as spacetimedb::rt::IntoReducerResult>::into_result;)*
134            }
135        };
136        impl #func_name {
137            fn invoke(__ctx: spacetimedb::ReducerContext, __args: &[u8]) -> spacetimedb::ReducerResult {
138                spacetimedb::rt::invoke_reducer(#func_name, __ctx, __args)
139            }
140        }
141        #[automatically_derived]
142        impl spacetimedb::rt::ReducerInfo for #func_name {
143            const NAME: &'static str = #reducer_name;
144            #(const LIFECYCLE: Option<spacetimedb::rt::LifecycleReducer> = Some(#lifecycle);)*
145            const ARG_NAMES: &'static [Option<&'static str>] = &[#(#opt_arg_names),*];
146            const INVOKE: spacetimedb::rt::ReducerFn = #func_name::invoke;
147        }
148    })
149}
