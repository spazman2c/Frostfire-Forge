1use crate::algebraic_value::AlgebraicValue;
2use crate::sum_type::SumType;
3
4/// A value of a sum type choosing a specific variant of the type.
5#[derive(Debug, Clone, Eq, PartialEq, Ord, PartialOrd, Hash)]
6pub struct SumValue {
7    /// A tag representing the choice of one variant of the sum type's variants.
8    pub tag: u8,
9    /// Given a variant `Var(Ty)` in a sum type `{ Var(Ty), ... }`,
10    /// this provides the `value` for `Ty`.
11    pub value: Box<AlgebraicValue>,
12}
13
14impl crate::Value for SumValue {
15    type Type = SumType;
16}
17
18impl SumValue {
19    /// Returns a new `SumValue` with the given `tag` and `value`.
20    pub fn new(tag: u8, value: impl Into<AlgebraicValue>) -> Self {
21        let value = Box::from(value.into());
22        Self { tag, value }
23    }
24
25    /// Returns a new `SumValue` with the given `tag` and unit value.
26    pub fn new_simple(tag: u8) -> Self {
27        Self::new(tag, ())
28    }
29}
30
31/// The tag of a `SumValue`.
32/// Can be used to read out the tag of a sum value without reading the payload.
33#[derive(Debug, Copy, Clone, Eq, PartialEq, Ord, PartialOrd, Hash)]
34pub struct SumTag(pub u8);
35
36impl From<SumTag> for SumValue {
37    fn from(SumTag(tag): SumTag) -> Self {
38        SumValue::new_simple(tag)
39    }
40}
