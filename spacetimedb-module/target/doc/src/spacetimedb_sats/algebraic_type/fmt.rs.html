6/// Wraps the algebraic `ty` into a `Display`able.
7///
8/// NOTE: You might ask: Why do we have a formatter and a notation for
9/// `AlgebraicType`s if we don't have an encoding for `AlgebraicType`s?
10///
11/// This is because we just want an easier to read text format for algebraic
12/// types. This could just as easily take in an algebraic value, which
13/// represents an algebraic type and format it that way. It's just more
14/// convenient to format it from the Rust type.
15pub fn fmt_algebraic_type(ty: &AlgebraicType) -> impl '_ + Display {
16    fmt_fn(move |f| match ty {
17        AlgebraicType::Ref(r) => write!(f, "{r}"),
18        AlgebraicType::Sum(ty) => write!(f, "{}", fmt_sum_type(ty)),
19        AlgebraicType::Product(ty) => write!(f, "{}", fmt_product_type(ty)),
20        AlgebraicType::Array(a) => write!(f, "Array<{}>", fmt(&a.elem_ty)),
21        AlgebraicType::Bool => write!(f, "Bool"),
22        AlgebraicType::I8 => write!(f, "I8"),
23        AlgebraicType::U8 => write!(f, "U8"),
24        AlgebraicType::I16 => write!(f, "I16"),
25        AlgebraicType::U16 => write!(f, "U16"),
26        AlgebraicType::I32 => write!(f, "I32"),
27        AlgebraicType::U32 => write!(f, "U32"),
28        AlgebraicType::I64 => write!(f, "I64"),
29        AlgebraicType::U64 => write!(f, "U64"),
30        AlgebraicType::I128 => write!(f, "I128"),
31        AlgebraicType::U128 => write!(f, "U128"),
32        AlgebraicType::I256 => write!(f, "I256"),
33        AlgebraicType::U256 => write!(f, "U256"),
34        AlgebraicType::F32 => write!(f, "F32"),
35        AlgebraicType::F64 => write!(f, "F64"),
36        AlgebraicType::String => write!(f, "String"),
37    })
38}

40/// Wraps the builtin `ty` into a `Display`able.
41pub fn fmt_product_type(ty: &ProductType) -> impl '_ + Display {
42    fmt_fn(move |f| {
43        write!(f, "(")?;
44        for (i, e) in ty.elements.iter().enumerate() {
45            if let Some(name) = &e.name {
46                write!(f, "{name}")?;
47            } else {
48                write!(f, "{i}")?;
49            }
50            write!(f, ": ")?;
51            write!(f, "{}", fmt(&e.algebraic_type))?;
52            if i < ty.elements.len() - 1 {
53                write!(f, ", ")?;
54            }
55        }
56        write!(f, ")")
57    })
58}

60/// Wraps the builtin `ty` into a `Display`able.
61fn fmt_sum_type(ty: &SumType) -> impl '_ + Display {
62    fmt_fn(move |f| {
63        if ty.variants.is_empty() {
64            return write!(f, "(|)");
65        }
66        write!(f, "(")?;
67        for (i, e) in ty.variants.iter().enumerate() {
68            if let Some(name) = &e.name {
69                write!(f, "{name}")?;
70                write!(f, ": ")?;
71            }
72            write!(f, "{}", fmt(&e.algebraic_type))?;
73            if i < ty.variants.len() - 1 {
74                write!(f, " | ")?;
75            }
76        }
77        write!(f, ")")
78    })
