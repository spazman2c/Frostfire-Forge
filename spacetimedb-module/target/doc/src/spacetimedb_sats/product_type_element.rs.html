1use crate::meta_type::MetaType;
2use crate::{AlgebraicType, SpacetimeType, WithTypespace};
3
4/// A factor / element of a product type.
5///
6/// An element consist of an optional name and a type.
7///
8/// NOTE: Each element has an implicit element tag based on its order.
9/// Uniquely identifies an element similarly to protobuf tags.
10#[derive(Debug, Clone, Eq, PartialEq, Ord, PartialOrd, Hash, SpacetimeType)]
11#[sats(crate = crate)]
12pub struct ProductTypeElement {
13    /// The name of the field / element.
14    ///
15    /// As our type system is structural,
16    /// a type like `{ foo: U8 }`, where `foo: U8` is the `ProductTypeElement`,
17    /// is inequal to `{ bar: U8 }`, although their `algebraic_type`s (`U8`) match.
18    pub name: Option<Box<str>>,
19    /// The type of the element.
20    ///
21    /// Only values of this type can be stored in the element.
22    pub algebraic_type: AlgebraicType,
23}
24
25impl ProductTypeElement {
26    /// Returns an element with the given `name` and `algebraic_type`.
27    pub const fn new(algebraic_type: AlgebraicType, name: Option<Box<str>>) -> Self {
28        Self { algebraic_type, name }
29    }
30
31    /// Returns a named element with `name` and `algebraic_type`.
32    pub fn new_named(algebraic_type: AlgebraicType, name: impl Into<Box<str>>) -> Self {
33        Self::new(algebraic_type, Some(name.into()))
34    }
35
36    /// Returns the name of the field.
37    pub fn name(&self) -> Option<&str> {
38        self.name.as_deref()
39    }
40
41    /// Returns whether the field has the given name.
42    pub fn has_name(&self, name: &str) -> bool {
43        self.name() == Some(name)
44    }
45}
46
47impl MetaType for ProductTypeElement {
48    fn meta_type() -> AlgebraicType {
49        AlgebraicType::product([
50            ("name", AlgebraicType::option(AlgebraicType::String)),
51            ("algebraic_type", AlgebraicType::ZERO_REF),
52        ])
53    }
54}
55
56impl From<AlgebraicType> for ProductTypeElement {
57    fn from(value: AlgebraicType) -> Self {
58        ProductTypeElement::new(value, None)
59    }
60}
61
62impl<'a> WithTypespace<'a, ProductTypeElement> {
63    #[inline]
64    pub fn algebraic_type(&self) -> WithTypespace<'a, AlgebraicType> {
65        self.with(&self.ty().algebraic_type)
66    }
