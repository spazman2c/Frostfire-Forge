5/// A variant of a sum type.
6///
7/// NOTE: Each element has an implicit element tag based on its order.
8/// Uniquely identifies an element similarly to protobuf tags.
9#[derive(Debug, Clone, Eq, PartialEq, Ord, PartialOrd, Hash, SpacetimeType)]
10#[sats(crate = crate)]
11pub struct SumTypeVariant {
12    /// The name of the variant, if any.
13    pub name: Option<Box<str>>,
14    /// The type of the variant.
15    ///
16    /// Unlike a language like Rust,
17    /// where we can have `enum _ { V1 { foo: A, bar: B, .. }, .. }`,
18    /// the product within the variant `V1`, i.e., `{ foo: A, bar: B, .. }`
19    /// is separated out in SATS into a separate product type.
20    /// So we would express this as `{ V1({ foo: A, bar: B, .. }), .. }`.
21    pub algebraic_type: AlgebraicType,
22}
23
24impl SumTypeVariant {
25    /// Returns a sum type variant with an optional `name` and `algebraic_type`.
26    pub const fn new(algebraic_type: AlgebraicType, name: Option<Box<str>>) -> Self {
27        Self { algebraic_type, name }
28    }
29
30    /// Returns a sum type variant with `name` and `algebraic_type`.
31    pub fn new_named(algebraic_type: AlgebraicType, name: impl AsRef<str>) -> Self {
32        Self {
33            algebraic_type,
34            name: Some(name.as_ref().into()),
35        }
36    }
37
38    /// Returns a unit variant with `name`.
39    pub fn unit(name: &str) -> Self {
40        Self::new_named(AlgebraicType::unit(), name)
41    }
42
43    /// Returns the name of the variant.
44    pub fn name(&self) -> Option<&str> {
45        self.name.as_deref()
46    }
47
48    /// Returns whether the variant has the given name.
49    pub fn has_name(&self, name: &str) -> bool {
50        self.name() == Some(name)
51    }
52
53    /// Returns whether this is a unit variant.
54    pub fn is_unit(&self) -> bool {
55        self.algebraic_type == AlgebraicType::unit()
56    }
57}
58
59impl MetaType for SumTypeVariant {
60    fn meta_type() -> AlgebraicType {
61        AlgebraicType::product([
62            ("name", AlgebraicType::option(AlgebraicType::String)),
63            ("algebraic_type", AlgebraicType::ZERO_REF),
64        ])
65    }
66}
67
68impl From<AlgebraicType> for SumTypeVariant {
69    fn from(algebraic_type: AlgebraicType) -> Self {
70        Self::new(algebraic_type, None)
71    }
