1/// Metrics collected during the course of a transaction.
2#[derive(Debug, Default, Copy, Clone)]
3pub struct ExecutionMetrics {
4    /// How many times is an index probed?
5    ///
6    /// Note that a single btree scan may return many values,
7    /// but will only result in a single index seek.
8    pub index_seeks: usize,
9    /// How many rows are iterated over?
10    ///
11    /// It is independent of the number of rows returned.
12    /// A query for example may return a single row,
13    /// but if it scans the entire table to find that row,
14    /// this metric will reflect that.
15    pub rows_scanned: usize,
16    /// How many bytes are read?
17    ///
18    /// This metric is incremented anytime we dereference a `RowPointer`.
19    ///
20    /// For reducers this happens at the WASM boundary,
21    /// when serializing entire rows via the BSATN encoding.
22    ///
23    /// In addition to the same BSATN serialization of the output rows,
24    /// queries will dereference a `RowPointer` for column projections.
25    /// Such is the case for filters as well as index and hash joins.
26    ///
27    /// One place where this metric is not tracked is index scans.
28    /// Specifically the key comparisons that occur during the scan.
29    pub bytes_scanned: usize,
30    /// How many bytes are written?
31    ///
32    /// Note, this is the same as bytes inserted,
33    /// because deletes just update a free list in the datastore.
34    /// They don't actually write or clear page memory.
35    pub bytes_written: usize,
36    /// How many bytes did we send to clients?
37    ///
38    /// This is not necessarily the same as bytes scanned,
39    /// since a single query may send bytes to multiple clients.
40    ///
41    /// In general, these are BSATN bytes, but JSON is also possible.
42    pub bytes_sent_to_clients: usize,
43    /// How many rows were inserted?
44    pub rows_inserted: u64,
45    /// How many rows were deleted?
46    pub rows_deleted: u64,
47    /// How many rows were updated?
48    pub rows_updated: u64,
49    /// How many subscription updates did we execute?
50    pub delta_queries_evaluated: u64,
51    /// How many subscriptions had some updates?
52    pub delta_queries_matched: u64,
53    /// How many times do we evaluate the same row in a subscription update?
54    pub duplicate_rows_evaluated: u64,
55    /// How many duplicate rows do we send in a subscription update?
56    pub duplicate_rows_sent: u64,
57}
58
59impl ExecutionMetrics {
60    pub fn merge(
61        &mut self,
62        ExecutionMetrics {
63            index_seeks,
64            rows_scanned,
65            bytes_scanned,
66            bytes_written,
67            bytes_sent_to_clients,
68            rows_inserted,
69            rows_deleted,
70            rows_updated,
71            delta_queries_evaluated,
72            delta_queries_matched,
73            duplicate_rows_evaluated,
74            duplicate_rows_sent,
75        }: ExecutionMetrics,
76    ) {
77        self.index_seeks += index_seeks;
78        self.rows_scanned += rows_scanned;
79        self.bytes_scanned += bytes_scanned;
80        self.bytes_written += bytes_written;
81        self.bytes_sent_to_clients += bytes_sent_to_clients;
82        self.rows_inserted += rows_inserted;
83        self.rows_deleted += rows_deleted;
84        self.rows_updated += rows_updated;
85        self.delta_queries_evaluated += delta_queries_evaluated;
86        self.delta_queries_matched += delta_queries_matched;
87        self.duplicate_rows_evaluated += duplicate_rows_evaluated;
88        self.duplicate_rows_sent += duplicate_rows_sent;
89    }
90}

92#[cfg(test)]
93mod tests {
94    use super::ExecutionMetrics;

96    #[test]
97    fn test_merge() {
98        let mut a = ExecutionMetrics::default();
99
100        a.merge(ExecutionMetrics {
101            index_seeks: 1,
102            rows_scanned: 1,
103            bytes_scanned: 1,
104            bytes_written: 1,
105            bytes_sent_to_clients: 1,
106            rows_inserted: 1,
107            rows_deleted: 1,
108            rows_updated: 1,
109            delta_queries_evaluated: 2,
110            delta_queries_matched: 3,
111            duplicate_rows_evaluated: 4,
112            duplicate_rows_sent: 2,
113        });
114
115        assert_eq!(a.index_seeks, 1);
116        assert_eq!(a.rows_scanned, 1);
117        assert_eq!(a.bytes_scanned, 1);
118        assert_eq!(a.bytes_written, 1);
119        assert_eq!(a.bytes_sent_to_clients, 1);
120        assert_eq!(a.rows_inserted, 1);
121        assert_eq!(a.rows_deleted, 1);
122        assert_eq!(a.delta_queries_evaluated, 2);
123        assert_eq!(a.delta_queries_matched, 3);
124        assert_eq!(a.duplicate_rows_evaluated, 4);

126    }
