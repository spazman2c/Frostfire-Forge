1//! Hashing.
2//!
3//! This module provides hashing for primitive floating-point values. Given the
4//! set of zero representations $Z$ and set of `NaN` representations $N$,
5//! hashing coalesces their representations such that:
6//!
7//! $$
8//! \begin{aligned}
9//! h(a)=h(b)&\mid a\in{Z},~b\in{Z}\cr\[1em\]
10//! h(a)=h(b)&\mid a\in{N},~b\in{N}
11//! \end{aligned}
12//! $$
13//!
14//! The `FloatHash` trait agrees with the ordering and equivalence relations of
15//! the `FloatOrd` and `FloatEq` traits.
16
17use core::hash::{Hash, Hasher};
18
19use crate::canonical::ToCanonicalBits;
20use crate::primitive::Primitive;
21use crate::Float;
22
23/// Hashing for primitive floating-point values.
24pub trait FloatHash {
25    // The name `hash` is ambiguous. `float_hash` is more clear and is
26    // consistent with the `FloatEq` and `FloatOrd` traits.
27    #[deprecated(since = "0.3.1", note = "use `FloatHash::float_hash` instead")]
28    fn hash<H>(&self, state: &mut H)
29    where
30        H: Hasher;
31
32    fn float_hash<H>(&self, state: &mut H)
33    where
34        H: Hasher,
35    {
36        // For compatibility, `float_hash` provides a default implementation.
37        // Eventually, `hash` will be removed along with this default
38        // implementation.
39        #[allow(deprecated)]
40        self.hash(state)
41    }
42}
43
44impl<T> FloatHash for T
45where
46    T: Float + Primitive,
47{
48    fn hash<H>(&self, state: &mut H)
49    where
50        H: Hasher,
51    {
52        self.to_canonical_bits().hash(state);
53    }
54}
55
56impl<T> FloatHash for [T]
57where
58    T: Float + Primitive,
59{
60    fn hash<H>(&self, state: &mut H)
61    where
62        H: Hasher,
63    {
64        for item in self.iter() {
65            item.float_hash(state);
66        }
67    }
68}
